```                                   
# --- Конфигурация ---
SOURCE_DIR="/home/vagrant/" # Исходная директория (обязательно с завершающим слэшем)
REMOTE_USER="vagrant"      # Пользователь на удаленном сервере
REMOTE_HOST="192.168.56.11" # IP или домен удаленного сервера
REMOTE_BASE_DIR="/tmp" # Базовая директория для бэкапов на удаленном сервере
KEEP_BACKUPS=5                 # Количество сохраняемых резервных копий
# --------------------
```

restore_manager.sh    
```
#!/bin/bash

# --- Конфигурация ---
RESTORE_DEST_DIR="/home/vagrant/" # Локальная директория для восстановления
REMOTE_USER="vagrant"                   # Пользователь на удаленном сервере
REMOTE_HOST="192.168.56.11"              # IP или домен удаленного сервера
REMOTE_BASE_DIR="/tmp"          # Базовая директория бэкапов на удаленном сервере
# --------------------

echo "Получение списка доступных резервных копий с удаленного сервера..."

# Получение списка бэкапов, исключая символическую ссылку 'latest'
BACKUPS_LIST=$(ssh "$REMOTE_USER@$REMOTE_HOST" "ls -td $REMOTE_BASE_DIR/*/ | grep -v '/latest/'")
if [ -z "$BACKUPS_LIST" ]; then
    echo "Резервные копии не найдены на удаленном сервере."
    exit 1
fi

# Преобразование списка в массив для удобного выбора
IFS=$'\n' read -d '' -r -a BACKUP_ARRAY <<< "$BACKUPS_LIST"

echo "Доступные резервные копии:"
for i in "${!BACKUP_ARRAY[@]}"; do
    echo "[$i] ${BACKUP_ARRAY[$i]}"
done

# Запрос выбора у пользователя
read -p "Введите номер копии для восстановления (например, 0 для самой свежей): " SELECTION_INDEX

# Проверка корректности ввода
if ! [[ "$SELECTION_INDEX" =~ ^[0-9]+$ ]] || [ "$SELECTION_INDEX" -ge "${#BACKUP_ARRAY[@]}" ]; then
    echo "Неверный выбор. Выход."
    exit 1
fi

SELECTED_BACKUP_PATH=${BACKUP_ARRAY[$SELECTION_INDEX]}
REMOTE_BACKUP_PATH_REL=$(basename "$SELECTED_BACKUP_PATH")
REMOTE_FULL_PATH="$REMOTE_BASE_DIR/$REMOTE_BACKUP_PATH_REL"

echo "Выбрана копия: $SELECTED_BACKUP_PATH"
echo "Восстановление в локальную директорию: $RESTORE_DEST_DIR"

# Подтверждение действия
read -p "Вы уверены, что хотите восстановить данные? Все файлы в $RESTORE_DEST_DIR будут синхронизированы с бэкапом. (y/N): " CONFIRM
if [[ "$CONFIRM" != "y" && "$CONFIRM" != "Y" ]]; then
    echo "Восстановление отменено."
    exit 0
fi

# Выполнение восстановления с помощью rsync
rsync -avzh --delete "$REMOTE_USER@$REMOTE_HOST:$REMOTE_FULL_PATH/" "$RESTORE_DEST_DIR"

if [ $? -eq 0 ]; then
    echo "Восстановление из резервной копии успешно завершено."
else
    echo "Ошибка при восстановлении данных."
    exit 1
fi
```
